create table public.class_subjects (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  subject_id uuid null,
  class_id uuid null,
  teacher_id uuid null,
  constraint class_subjects_pkey primary key (id),
  constraint class_subjects_class_id_fkey foreign KEY (class_id) references classes (id),
  constraint class_subjects_subject_id_fkey foreign KEY (subject_id) references subjects (id),
  constraint class_subjects_teacher_id_fkey foreign KEY (teacher_id) references teachers (id)
) TABLESPACE pg_default;

create table public.classes (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default now(),
  class_name text null,
  section text null,
  school_id uuid null,
  constraint classes_pkey primary key (id),
  constraint classes_school_id_fkey foreign KEY (school_id) references schools (id)
) TABLESPACE pg_default;

create table public.schools (
  id uuid not null default gen_random_uuid (),
  user_id uuid null,
  name text null,
  created_at timestamp without time zone null default now(),
  username text null,
  constraint schools_pkey primary key (id),
  constraint schools_user_id_fkey foreign KEY (user_id) references users (id)
) TABLESPACE pg_default;

create table public.student_records (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default now(),
  student_id uuid null,
  class_id uuid null,
  academic_year text null,
  is_current boolean null default true,
  constraint students_enrollments_pkey primary key (id),
  constraint student_records_class_id_fkey foreign KEY (class_id) references classes (id),
  constraint student_records_student_id_fkey foreign KEY (student_id) references students (id) on delete CASCADE
) TABLESPACE pg_default;

create table public.students (
  id uuid not null default gen_random_uuid (),
  user_id uuid null,
  school_id uuid null,
  name text null,
  created_at timestamp without time zone null default now(),
  constraint students_pkey primary key (id),
  constraint students_school_id_fkey foreign KEY (school_id) references schools (id),
  constraint students_user_id_fkey foreign KEY (user_id) references users (id) on delete CASCADE
) TABLESPACE pg_default;

create table public.subjects (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default now(),
  school_id uuid null,
  subject_name text null,
  constraint subjects_pkey primary key (id),
  constraint subjects_school_id_fkey foreign KEY (school_id) references schools (id)
) TABLESPACE pg_default;

create table public.teachers (
  id uuid not null default gen_random_uuid (),
  user_id uuid null,
  school_id uuid null,
  name text null,
  created_at timestamp without time zone null default now(),
  constraint teachers_pkey primary key (id),
  constraint teachers_school_id_fkey foreign KEY (school_id) references schools (id),
  constraint teachers_user_id_fkey foreign KEY (user_id) references users (id)
) TABLESPACE pg_default;

create table public.users (
  id uuid not null,
  email text null,
  role public.role null,
  created_at timestamp without time zone null default now(),
  constraint users_pkey primary key (id)
) TABLESPACE pg_default;

create table public.quiz (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default now(),
  school_id uuid null,
  class_id uuid null,
  teacher_id uuid null,
  title text null,
  count bigint null,
  subject_id uuid null,
  constraint quiz_pkey primary key (id),
  constraint quiz_class_id_fkey foreign KEY (class_id) references classes (id),
  constraint quiz_school_id_fkey foreign KEY (school_id) references schools (id),
  constraint quiz_subject_id_fkey foreign KEY (subject_id) references subjects (id),
  constraint quiz_teacher_id_fkey foreign KEY (teacher_id) references teachers (id)
) TABLESPACE pg_default;

create table public.questions (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default now(),
  question text null,
  options json null,
  answer text null,
  quiz_id uuid null,
  constraint questions_pkey primary key (id),
  constraint questions_quiz_id_fkey foreign KEY (quiz_id) references quiz (id) on delete CASCADE
) TABLESPACE pg_default;

create table public.quiz_attempts (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default now(),
  quiz_id uuid not null,
  student_id uuid not null,
  score numeric null,
  correct_count numeric null,
  xp_earned integer not null default 0,
  constraint quiz_attempts_pkey primary key (id),
  constraint quiz_attempts_unique unique (quiz_id, student_id),
  constraint quiz_attempts_quiz_id_fkey foreign KEY (quiz_id) references quiz (id) on delete CASCADE,
  constraint quiz_attempts_student_id_fkey foreign KEY (student_id) references students (id) on delete CASCADE
) TABLESPACE pg_default;

create table public.quiz_answers (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default now(),
  attempt_id uuid not null,
  question_id uuid not null,
  selected_option text null,
  is_correct boolean null,
  constraint quiz_answers_pkey primary key (id),
  constraint quiz_answers_unique unique (attempt_id, question_id),
  constraint quiz_answers_attempt_id_fkey foreign KEY (attempt_id) references quiz_attempts (id) on delete CASCADE,
  constraint quiz_answers_question_id_fkey foreign KEY (question_id) references questions (id) on delete CASCADE
) TABLESPACE pg_default;

create table public.xp_logs (
  id uuid not null default gen_random_uuid (),
  student_id uuid not null,
  school_id uuid not null,
  class_id uuid null,
  source text not null,
  source_id uuid null,
  xp integer not null,
  metadata jsonb null,
  created_at timestamp with time zone null default now(),
  constraint xp_logs_pkey primary key (id),
  constraint xp_logs_class_id_fkey foreign KEY (class_id) references classes (id),
  constraint xp_logs_school_id_fkey foreign KEY (school_id) references schools (id) on delete CASCADE,
  constraint xp_logs_student_id_fkey foreign KEY (student_id) references students (id) on delete CASCADE,
  constraint xp_logs_xp_check check ((xp > 0))
) TABLESPACE pg_default;

create table public.student_xp (

  constraint student_xp_school_id_fkey foreign KEY (school_id) references schools (id) on delete CASCADE,
  constraint student_xp_student_id_fkey foreign KEY (student_id) references students (id) on delete CASCADE
) TABLESPACE pg_default;
